public int roundSum(int a, int b, int c)
{
  int newA = round10(a);
  int newB = round10(b);
  int newC = round10(c);
  
  return newA+newB+newC;
}

public int round10(int num)
{
  int rounded = num;
  if(num%10>=5)
    rounded = num+(10-num%10);
  else
    rounded = num-(num%10);
  
  return rounded;
}
